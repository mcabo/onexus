033
034import javax.servlet.ServletException;
035import javax.servlet.http.HttpServlet;
036import javax.servlet.http.HttpServletRequest;
037import javax.servlet.http.HttpServletResponse;
038import java.io.BufferedReader;
039import java.io.IOException;
040import java.io.PrintWriter;
041import java.util.Iterator;
042import java.util.List;
043import java.util.Map;
044
045public class OqlServlet extends HttpServlet {
046
047    private static final Logger log = LoggerFactory.getLogger(OqlServlet.class);
048    private IResourceManager resourceManager;
049    private ICollectionManager collectionManager;
050    private IQueryParser queryParser;
051
052
053    @Override
054    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
055
056        String oqlQuery = req.getParameter("query");
057        String filename = req.getParameter("filename");
058
059        try {
060            response(resp, oqlQuery, filename);
061        } catch (Exception e) {
062            resp.sendError(HttpServletResponse.SC_NOT_FOUND, e.getMessage());
063        }
064    }
065
066    @Override
067    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
068
069        try {
070            BufferedReader reader = new BufferedReader(req.getReader());
071
072            StringBuilder sb = new StringBuilder();
073            String line = reader.readLine();
074            while (line != null) {
075                sb.append(line + "\n");
076                line = reader.readLine();
077            }
078            reader.close();
079            if (sb.length() > 0) {
080                response(resp, sb.toString(), null);
081            } else {
082                resp.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, "Missing query");
083            }
084
085        } catch (Exception e) {
086            log.error("Query service", e);
087            resp.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, e.getMessage());
088        }
089
090    }
091
092    public void response(HttpServletResponse resp, String oqlQuery, String filename) throws ServletException, IOException {
093
094        if (filename != null) {
095            resp.setHeader("Content-Disposition", "attachment; filename=\"" + filename + "\"");
096        }
097
098        PrintWriter pw = resp.getWriter();
099
100        Query query;
101        try {
102            query = queryParser.parseQuery(oqlQuery);
103        } catch (Exception e) {
104            throw new OnexusException("Error parsing OQL. " + oqlQuery, e);
105        }
106
107        IEntityTable result;
108        try {
109            result = collectionManager.load(query);
110        } catch (Exception e) {
111            throw new OnexusException("Error loading OQL. " + oqlQuery, e);
112        }
113
114        writeHeader(pw, result);
115
116        while (result.next()) {
117            writeRow(pw, result);
118        }
119
120        result.close();
121        pw.close();
122        resp.flushBuffer();
123    }
124
125    private void writeHeader(PrintWriter response, IEntityTable table) {
126
127        Iterator<Map.Entry<String, List<String>>> selectIt = table.getQuery().getSelect().entrySet().iterator();
128        while (selectIt.hasNext()) {
129            Map.Entry<String, List<String>> select = selectIt.next();
130
131            ORI collectionUri = QueryUtils.getCollectionOri(table.getQuery(), select.getKey());
132            Collection collection = resourceManager.load(Collection.class, collectionUri);
133
134            Iterator<String> fieldId = select.getValue().iterator();
135            while (fieldId.hasNext()) {
136                Field field = collection.getField(fieldId.next());
137                if (field == null) {
138                    continue;
139                }
140                String label = field.getLabel();
141                if (label == null) {
142                    label = field.getId();
143                }
144                response.write(label);
145
146                if (fieldId.hasNext() || selectIt.hasNext()) {
147                    response.write("\t");
148                }
149            }
150
151        }
152
153
154        response.write("\n");
155    }
156
157    private static void writeRow(PrintWriter response, IEntityTable table) {
158
159        Iterator<Map.Entry<String, List<String>>> selectIt = table.getQuery().getSelect().entrySet().iterator();
160        while (selectIt.hasNext()) {
161            Map.Entry<String, List<String>> select = selectIt.next();
162
163            ORI collection = QueryUtils.getCollectionOri(table.getQuery(), select.getKey());
164            IEntity entity = table.getEntity(collection);
165
166            Iterator<String> fieldId = select.getValue().iterator();
167            while (fieldId.hasNext()) {
168                response.write(String.valueOf(entity.get(fieldId.next())));
169                if (fieldId.hasNext() || selectIt.hasNext()) {
170                    response.write("\t");
171                }
172            }
173        }
174
175        response.write("\n");
176    }

196    }

199        this.queryParser = queryParser;
200    }
