077    @Override
078    public boolean isLinkable(Query query, ORI collectionUri) {
079
080        Collection joinCollection = resourceManager.load(Collection.class, collectionUri);
081
082        for (Map.Entry<String, ORI> tpDefine : query.getDefine().entrySet()) {
083
084            ORI tpCollectionUri = tpDefine.getValue().toAbsolute(query.getOn());
085
086            Collection tpJoinCollection = resourceManager.load(Collection.class, tpCollectionUri);
087
088            if (joinCollection.equals(tpJoinCollection)) {
089                return true;
090            }
091
092            List<FieldLink> links = LinkUtils.getLinkFields(query.getOn(), joinCollection, tpJoinCollection);
093
094            if (links != null && !links.isEmpty()) {
095                return true;
096            }
097        }
098
099        return false;
100    }

102    @Override
103    public IEntityTable load(Query query) {
104        LOGGER.debug("Loading query {}", query);
105
106        Set<ORI> notRegisteredCollections = new HashSet<ORI>();
107        for (ORI collectionURI : getQueryCollections(query)) {
108            if (!collectionStore.isRegistered(collectionURI)) {
109                notRegisteredCollections.add(collectionURI);
110            }
111        }
112
113        String taskId = Integer.toHexString(query.hashCode());
114        Progress progress = getTask(taskId);
115
116        if (progress == null && !notRegisteredCollections.isEmpty()) {
117            progress = new Progress(taskId, "Loading collections");
118
119            LOGGER.info("Starting task {}", taskId);
120            for (ORI collectionURI : notRegisteredCollections) {
121
122                Project project = resourceManager.getProject(collectionURI.getProjectUrl());
123                Collection collection = resourceManager.load(Collection.class, collectionURI);
124
125                if (collection == null) {
126                    progress.error("Unknown collection '" + collectionURI + "'");
127                    progress.fail();
128                } else {
129
130                    String subTaskId = Integer.toHexString(collectionURI.hashCode());
131
132                    if (!runningCollections.containsKey(subTaskId)) {
133
134                        Progress subProgress = new Progress(subTaskId, "Load '" + collectionURI.getPath() + "'");
135                        progressManager.addProgress(subProgress);
136                        runningCollections.put(subTaskId, subProgress);
137
138                        LOGGER.info("Registering {}", collectionURI);
139                        collectionStore.register(collectionURI);
140
141                        LOGGER.info("Submiting store collection '{}'", collectionURI);
142
143                        Loader loader = collection.getLoader();
144                        Plugin plugin = project.getPlugin(loader.getPlugin());
145                        ICollectionLoader collectionLoader = resourceManager.getLoader(ICollectionLoader.class, plugin, loader);
146
147                        Runnable command = new InsertCollectionRunnable(runningCollections, plugin, collection, collectionLoader, collectionStore);
148                        executorService.submit(command);
149
150                    }
151
152                    progress.addSubTask(runningCollections.get(subTaskId));
153                }
154
155            }
156        }
157
158        IEntityTable partialResults = collectionStore.load(query);
159
160        if (progress != null) {
161            partialResults = new ProgressEntityTable(progress, partialResults);
162        }
163
164        return partialResults;
165    }
166
167    private Set<ORI> getQueryCollections(Query query) {
168
169        Set<ORI> queryCollections = new HashSet<ORI>();
170        ORI onUri = query.getOn();
171
172        for (ORI collectionUri : query.getDefine().values()) {
173            queryCollections.add(collectionUri.toAbsolute(onUri));
174        }
175
176        return queryCollections;
177    }
178
179    public Progress getTask(String taskId) {
180        Progress progress = runningTasks.get(taskId);
181
182        if (progress == null) {
183            return null;
184        }
185
186        if (progress.isDone()) {
187            runningTasks.remove(taskId);
188        }
189
190        return progress;
191    }
192
193    @Override
194    public void unload(ORI collectionURI) {
195
196        if (collectionStore.isRegistered(collectionURI)) {
197            // If in future exist multiple collectionStores this action makes
198            // not sense!
199            collectionStore.deregister(collectionURI);
200        }
201    }
