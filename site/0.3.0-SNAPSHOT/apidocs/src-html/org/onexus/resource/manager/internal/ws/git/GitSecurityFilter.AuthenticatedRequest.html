050public class GitSecurityFilter implements Filter {
051
052    private static final Logger log = LoggerFactory.getLogger(GitSecurityFilter.class);
053
054    private static final String BASIC = "Basic";
055    private static final String CHALLENGE = BASIC + " realm=\"karaf\"";
056    private static final String APPLICATION_POLICY_NAME = "karaf";
057
058    private static final String gitReceivePack = "/git-receive-pack";
059    private static final String gitUploadPack = "/git-upload-pack";
060    private static final String[] suffixes = {gitReceivePack, gitUploadPack, "/info/refs", "/HEAD", "/objects"};
061
062    private IAuthorizationManager authorizationManager;
063    private IResourceManager resourceManager;
064
065
066    @Override
067    public void init(FilterConfig filterConfig) throws ServletException {
068        log.debug("FilterConfig: " + filterConfig);
069    }
070
071    @Override
072    public void doFilter(final ServletRequest request, final ServletResponse response,
073                         final FilterChain chain) throws IOException, ServletException {
074
075        HttpServletRequest httpRequest = (HttpServletRequest) request;
076        HttpServletResponse httpResponse = (HttpServletResponse) response;
077
078        String fullUrl = getFullUrl(httpRequest);
079        String repository = extractRepositoryName(fullUrl);
080        String fullSuffix = fullUrl.substring(repository.length());
081        String requestPrivilege = getUrlRequestAction(fullSuffix);
082        String user = getUser(httpRequest);
083        String serverUrl = getServerUrl(httpRequest);
084
085        ORI resource = new ORI(serverUrl + '/' + repository);
086        if (!authorizationManager.check(requestPrivilege, resource) && user == null) {
087            httpResponse.setHeader("WWW-Authenticate", CHALLENGE);
088            httpResponse.sendError(HttpServletResponse.SC_UNAUTHORIZED);
089            return;
090        }
091
092        Project project = null;
093        try {
094            project = resourceManager.getProject(resource.getProjectUrl());
095        } catch (Exception e) {
096            log.error("At git serving '" + resource + "'", e);
097        }
098
099        if (project == null && requestPrivilege.equals(IAuthorizationManager.WRITE)) {
100            //TODO automatic project creation on push
101        }
102
103        if (project == null) {
104            httpResponse.sendError(HttpServletResponse.SC_NOT_FOUND);
105            return;
106        }
107
108        chain.doFilter(new AuthenticatedRequest(user, httpRequest), httpResponse);
109
110    }
111
112    private String getServerUrl(HttpServletRequest httpRequest) {
113        //TODO
114        return "http://localhost:8181/git";
115    }
116
117    protected String getUser(HttpServletRequest httpRequest) {
118
119        final String authorization = httpRequest.getHeader("Authorization");
120
121        if (authorization != null && authorization.startsWith(BASIC)) {
122
123            // Authorization: Basic base64credentials
124            String base64Credentials = authorization.substring(BASIC.length()).trim();
125            String credentials = new String(Base64.decode(base64Credentials), Charset.forName("UTF-8"));
126
127            // credentials = username:password
128            final String[] values = credentials.split(":", 2);
129
130            if (values.length == 2) {
131                String username = values[0];
132                char[] password = values[1].toCharArray();
133                boolean valid = authenticate(username, new String(password));
134                if (valid) {
135                    return username;
136                }
137            }
138            log.info(MessageFormat.format("AUTH: invalid credentials ({0})", credentials));
139        }
140        return null;
141    }

143    @Override
144    public void destroy() {
145    }
146
147    private String getFullUrl(HttpServletRequest httpRequest) {
148        String servletUrl = httpRequest.getContextPath() + httpRequest.getServletPath();
149        String url = httpRequest.getRequestURI().substring(servletUrl.length());
150        String params = httpRequest.getQueryString();
151        if (url.length() > 0 && url.charAt(0) == '/') {
152            url = url.substring(1);
153        }
154        String fullUrl = url + (StringUtils.isEmptyOrNull(params) ? "" : ("?" + params));
155        return fullUrl;
156    }
157
158    private String extractRepositoryName(String url) {
159        String repository = url;
160        // get the repository name from the url by finding a known url suffix
161        for (String urlSuffix : suffixes) {
162            if (repository.indexOf(urlSuffix) > -1) {
163                repository = repository.substring(0, repository.indexOf(urlSuffix));
164            }
165        }
166        return repository;
167    }
168
169    private String getUrlRequestAction(String suffix) {
170        if (!StringUtils.isEmptyOrNull(suffix)) {
171            if (suffix.startsWith(gitReceivePack)) {
172                return IAuthorizationManager.WRITE;
173            } else if (suffix.startsWith(gitUploadPack)) {
174                return IAuthorizationManager.READ;
175            } else if (suffix.contains("?service=git-receive-pack")) {
176                return IAuthorizationManager.WRITE;
177            } else if (suffix.contains("?service=git-upload-pack")) {
178                return IAuthorizationManager.READ;
179            } else {
180                return IAuthorizationManager.READ;
181            }
182        }
183        return null;
184    }
185
186    public boolean authenticate(String username, String password) {
187        boolean authenticated;
188        LoginCallbackHandler handler = new LoginCallbackHandler(username, password);
189        try {
190            javax.security.auth.login.LoginContext javaCtx = new javax.security.auth.login.LoginContext(APPLICATION_POLICY_NAME, handler);
191            javaCtx.login();
192            authenticated = true;
193
194            LoginContext ctx = new LoginContext(username);
195
196            Subject subject = javaCtx.getSubject();
197            if (subject != null) {
198                for (Principal p : subject.getPrincipals()) {
199                    ctx.addRole(p.getName());
200                }
201            }
202            LoginContext.set(ctx);
203
204        } catch (LoginException e) {
205            authenticated = false;
206        }
207        return authenticated;
208    }

226    private class LoginCallbackHandler implements CallbackHandler {
227
228        private String username;
229        private String password;
230
231        public LoginCallbackHandler(String username, String password) {
232            this.username = username;
233            this.password = password;
234        }
235
236        public void handle(Callback[] callbacks) throws IOException, UnsupportedCallbackException {
237            for (int i = 0; i < callbacks.length; i++) {
238                Callback callback = callbacks[i];
239                if (callback instanceof NameCallback) {
240                    ((NameCallback) callback).setName(username);
241                } else if (callback instanceof PasswordCallback) {
242                    PasswordCallback pwCallback = (PasswordCallback) callback;
243                    pwCallback.setPassword(password.toCharArray());
244                } else {
245                    throw new UnsupportedCallbackException(callbacks[i], "Callback type not supported");
246                }
247            }
248        }
249    }

251    /**
252     * Wraps a standard HttpServletRequest and overrides user principal methods.
253     */
254    public static class AuthenticatedRequest extends ServletRequestWrapper {
255
256        private String user;
257
258        public AuthenticatedRequest(String user, HttpServletRequest req) {
259            super(req);
260            this.user = user;
261        }
262
263        @Override
264        public String getRemoteUser() {
265            return user;
266        }
267
268        @Override
269        public Principal getUserPrincipal() {
270            return new Principal() {
271                @Override
272                public String getName() {
273                    return user;
274                }
275            };
276        }
277    }
