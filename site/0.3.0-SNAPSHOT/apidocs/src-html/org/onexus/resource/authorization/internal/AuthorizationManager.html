018package org.onexus.resource.authorization.internal;
019
020import org.onexus.resource.api.IAuthorizationManager;
021import org.onexus.resource.api.LoginContext;
022import org.onexus.resource.api.ORI;
023
024import java.io.File;
025import java.io.FileInputStream;
026import java.io.FileOutputStream;
027import java.io.IOException;
028import java.util.Collections;
029import java.util.HashSet;
030import java.util.Properties;
031import java.util.Set;
032
033public class AuthorizationManager implements IAuthorizationManager {
034
035    public static String ANONYMOUS_USER = "anonymous";
036
037    private String authorizationFile;
038    private Properties properties;
039
040    @Override
041    public boolean check(String privilege, ORI resourceOri) {
042        return getPrivileges(resourceOri).contains(privilege);
043    }
044
045    @Override
046    public Set<String> getPrivileges(ORI resourceOri) {
047
048        LoginContext ctx = LoginContext.get();
049
050        if (ctx == null) {
051            return Collections.emptySet();
052        }
053
054        Set<String> result = new HashSet<String>();
055
056        // First load roles privileges
057        for (String role : ctx.getRoles()) {
058            result.addAll(loadPrivileges("role-" + role, resourceOri));
059        }
060
061        // Load user privileges
062        result.addAll(loadPrivileges(ctx.getUserName(), resourceOri));
063
064        return result;
065    }
066
067    private Set<String> loadPrivileges(String key, ORI resourceOri) {
068
069        Set<String> result = new HashSet<String>();
070
071        if (key == null) {
072            key = ANONYMOUS_USER;
073        }
074
075        String value = properties.getProperty(key);
076
077        if (value == null || value.isEmpty()) {
078            return result;
079        }
080
081        String items[] = value.split(",");
082        for (String item : items) {
083            String pair[] = item.split("\\|\\|");
084
085            String regex = pair[0].trim();
086            String privileges[] = pair[1].split("\\|");
087
088            if (resourceOri.toString().matches(regex)) {
089                for (String privilege : privileges) {
090                    result.add(privilege.trim().toLowerCase());
091                }
092            }
093        }
094
095        return result;
096    }
097
098    public void load() {
099        this.properties = new Properties();
100
101        try {
102
103            File file = new File(authorizationFile);
104
105            if (!file.exists()) {
106                this.properties.setProperty("role-admin", ".*||read|write|load|unload|grant");
107                this.properties.setProperty(ANONYMOUS_USER, ".*||read");
108                this.properties.setProperty("role-registered", ".*||read");
109                this.properties.store(new FileOutputStream(file), "Syntax: username = [regular expression to match against ORI] || [privilege 1] | [privilege 2] | ... , more...");
110            } else {
111                properties.load(new FileInputStream(authorizationFile));
112            }
113        } catch (IOException e) {
114            throw new IllegalStateException(e);
115        }
116    }
117
118    public String getAuthorizationFile() {
119        return authorizationFile;
120    }
121
122    public void setAuthorizationFile(String authorizationFile) {
123        this.authorizationFile = authorizationFile;
124    }
125}
