001package org.onexus.resource.api;
002
003import java.util.HashMap;
004import java.util.Map;
005
006public class Parameters extends HashMap<ParameterKey, String> {
007
008    public Parameters() {
009        super();
010    }
011
012    public Parameters(Map<ParameterKey, String> map) {
013        super(map);
014    }
015
016    public String get(Object key) {
017
018        if (!(key instanceof ParameterKey)) {
019            throw new UnsupportedOperationException("Key object must implement ParameterKey interface");
020        }
021
022        ParameterKey parameterKey = (ParameterKey) key;
023
024        if (!parameterKey.isOptional() && !containsKey(key)) {
025            throw new UnsupportedOperationException(
026                    "Parameter '" + parameterKey.getKey() + " - " + parameterKey.getDescription() + "' is mandatory");
027        }
028
029        String value = super.get(key);
030
031        if (value == null) {
032            value = parameterKey.getDefault();
033        }
034
035        return value;
036    }
037
038}
