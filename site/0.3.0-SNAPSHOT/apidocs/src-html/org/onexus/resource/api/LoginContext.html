018package org.onexus.resource.api;
019
020import java.io.Serializable;
021import java.util.HashSet;
022import java.util.Set;
023
024public class LoginContext implements Serializable {
025
026    private static ThreadLocal<LoginContext> loginContext = new ThreadLocal<LoginContext>() {
027        @Override
028        protected LoginContext initialValue() {
029            return new LoginContext();
030        }
031    };
032
033    public static LoginContext get() {
034        return loginContext.get();
035    }
036
037    public static void set(LoginContext ctx) {
038        loginContext.set(ctx);
039    }
040
041    private String userName;
042    private Set<String> roles = new HashSet<String>();
043
044    public LoginContext() {
045        this.userName = null;
046    }
047
048    public LoginContext(String userName) {
049        this.userName = userName;
050    }
051
052    public void logout() {
053        this.userName = null;
054    }
055
056    public boolean isAnonymous() {
057        return (userName == null);
058    }
059
060    public String getUserName() {
061        return userName;
062    }
063
064    public Set<String> getRoles() {
065        return roles;
066    }
067
068    public void addRole(String role) {
069        roles.add(role);
070    }
071
072    /**
073     * Whether this context object containes the provided role.
074     *
075     * @param role the role to check
076     * @return true if it contains the role, false otherwise
077     */
078    public boolean hasRole(final String role) {
079        if (userName != null && role != null) {
080            return roles.contains(role);
081        }
082        return false;
083    }
084
085
086}
