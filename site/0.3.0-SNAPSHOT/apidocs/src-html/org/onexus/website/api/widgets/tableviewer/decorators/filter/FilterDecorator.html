018package org.onexus.website.api.widgets.tableviewer.decorators.filter;
019
020import org.apache.wicket.AttributeModifier;
021import org.apache.wicket.ajax.AjaxRequestTarget;
022import org.apache.wicket.markup.html.WebMarkupContainer;
023import org.apache.wicket.model.IModel;
024import org.apache.wicket.model.Model;
025import org.onexus.collection.api.Field;
026import org.onexus.collection.api.IEntity;
027import org.onexus.resource.api.ORI;
028import org.onexus.website.api.Website;
029import org.onexus.website.api.events.AbstractEvent;
030import org.onexus.website.api.events.EventAddFilter;
031import org.onexus.website.api.pages.browser.BrowserPageLink;
032import org.onexus.website.api.pages.browser.BrowserPageStatus;
033import org.onexus.website.api.pages.browser.FilterEntity;
034import org.onexus.website.api.widgets.tableviewer.decorators.utils.FieldDecorator;
035import org.onexus.website.api.widgets.tableviewer.decorators.utils.LinkPanel;
036
037public class FilterDecorator extends FieldDecorator {
038
039    private ORI collectionId;
040
041    public FilterDecorator(ORI collectionId, Field field) {
042        super(field);
043        this.collectionId = collectionId;
044    }
045
046    @Override
047    public void populateCell(WebMarkupContainer cellContainer, String componentId, IModel<IEntity> data) {
048        Object value = getValue(data.getObject());
049
050        //TODO String label = getFormatValue(data.getObject());
051        //TODO String tooltip = (value == null ? "No data" : value.toString());
052        String label = "<i class=\"icon-hand-up\"></i>";
053        String tooltip = "Filter results by " + getField().getLabel() + " = " + String.valueOf(value);
054
055        cellContainer.add(new LinkPanel(componentId, label, getLink(collectionId, data, tooltip)));
056    }
057
058    protected WebMarkupContainer getLink(ORI collectionId, IModel<IEntity> data, String tooltip) {
059
060        String entityId = null;
061        IEntity entity = data.getObject();
062
063        if (entity.getCollection().getORI().equals(collectionId)) {
064            entityId = entity.getId();
065        } else {
066            entityId = String.valueOf(entity.get(getField().getId()));
067        }
068
069        WebMarkupContainer link = new BrowserPageLink<FilterEntity>(LinkPanel.LINK_ID, Model.of(new FilterEntity(collectionId, entityId))) {
070
071            @Override
072            public void onClick(AjaxRequestTarget target) {
073                BrowserPageStatus status = getBrowserPageStatus();
074                FilterEntity rowEntity = getModelObject();
075
076                // Try to make the link relative to the current website project
077                Website website = findParent(Website.class);
078                if (website != null) {
079                    ORI projectUri = website.getConfig().getORI().getParent();
080                    ORI relativeUri = rowEntity.getFilteredCollection().toRelative(projectUri);
081                    rowEntity.setFilteredCollection(relativeUri);
082                }
083
084                AbstractEvent[] events = FilterDecorator.this.onClick(rowEntity, status);
085                for (AbstractEvent event : events) {
086                    sendEvent(event);
087                }
088            }
089
090        };
091
092        link.add(new AttributeModifier("rel", "tooltip"));
093        link.add(new AttributeModifier("title", tooltip));
094        link.add(new AttributeModifier("data-placement", "right"));
095        return link;
096    }
097
098    protected AbstractEvent[] onClick(FilterEntity rowEntity, BrowserPageStatus status) {
099
100        // Fix current row entity
101        status.getFilters().add(rowEntity);
102
103        return new AbstractEvent[]{EventAddFilter.EVENT};
104    }
105
106
107}
