075    public SearchPage(String componentId, IModel<SearchPageStatus> statusModel) {
076        super(componentId, statusModel);
077
078        IModel<SearchPageStatus> pageStatusModel = new PropertyModel<SearchPageStatus>(this, "status");
079
080        Form form = new Form<SearchPageStatus>("form") {
081            @Override
082            protected void onSubmit() {
083                ORI baseUri = SearchPage.this.getConfig().getWebsiteConfig().getORI().getParent();
084                SearchPage.this.addOrReplace(new BoxesPanel("boxes", SearchPage.this.getStatus(), baseUri, null).setOutputMarkupId(true));
085            }
086        };
087
088        // By default use the first search type
089        List<SearchType> types = getConfig().getTypes();
090        if (getStatus().getType() == null && !types.isEmpty()) {
091            getStatus().setType(types.get(0));
092        }
093
094
095        final TextField<String> search = new TextField<String>("search", new PropertyModel<String>(pageStatusModel, "search"));
096        search.setOutputMarkupId(true);
097
098        search.add(new AutoCompleteBehavior<IEntity>(new EntityRenderer(), new AutoCompleteSettings()) {
099            @Override
100            protected Iterator<IEntity> getChoices(String input) {
101                return getAutocompleteChoices(input);
102            }
103        });
104        search.add(new AjaxFormComponentUpdatingBehavior("onchange") {
105            @Override
106            protected void onUpdate(AjaxRequestTarget target) {
107                ORI baseUri = SearchPage.this.getConfig().getWebsiteConfig().getORI().getParent();
108                SearchPage.this.addOrReplace(new BoxesPanel("boxes", SearchPage.this.getStatus(), baseUri, null).setOutputMarkupId(true));
109                target.add(SearchPage.this.get("boxes"));
110            }
111        });
112
113        form.add(search);
114
115        // Filters list modal
116        final WebMarkupContainer widgetModal = new WebMarkupContainer("widgetModal");
117        widgetModal.setOutputMarkupId(true);
118        widgetModal.add(new Label("header", ""));
119        widgetModal.add(new EmptyPanel("widget"));
120        widgetModal.add(new AjaxLink<String>("close") {
121            @Override
122            public void onClick(AjaxRequestTarget target) {
123                target.appendJavaScript("$('#" + widgetModal.getMarkupId() + "').modal('hide')");
124            }
125        });
126        form.add(widgetModal);
127        final AjaxLink<String> list = new AjaxLink<String>("list") {
128
129            @Override
130            public void onClick(AjaxRequestTarget target) {
131
132                FiltersWidgetConfig filters = getStatus().getType().getFilters();
133
134                if (filters!=null) {
135                    FiltersWidgetStatus status = filters.createEmptyStatus();
136                    status.setConfig(filters);
137                    setFiltersStatus(status);
138                    widgetModal.addOrReplace(new Label("header", filters.getTitle()));
139                    widgetModal.addOrReplace(new SearchFiltersWidget("widget", new PropertyModel<FiltersWidgetStatus>(SearchPage.this, "filtersStatus")) {
140
141                        @Override
142                        protected void applyFilter(FilterConfig filterConfig, AjaxRequestTarget target) {
143                            filterConfig.setDeletable(true);
144                            search.setModelValue(new String[] { filterConfig.getName() });
145                            target.add(search);
146                            ORI baseUri = SearchPage.this.getConfig().getWebsiteConfig().getORI().getParent();
147                            SearchPage.this.addOrReplace(new BoxesPanel("boxes", SearchPage.this.getStatus(), baseUri, filterConfig).setOutputMarkupId(true));
148                            target.add(SearchPage.this.get("boxes"));
149                            target.appendJavaScript("$('#" + widgetModal.getMarkupId() + "').modal('hide')");
150                        }
151                    });
152                    target.add(widgetModal);
153                    target.appendJavaScript("$('#" + widgetModal.getMarkupId() + "').modal('show')");
154                }
155            }
156        };
157        list.setOutputMarkupPlaceholderTag(true);
158        form.add(list);
159
160        FiltersWidgetConfig filters = getStatus().getType().getFilters();
161        if (filters == null) {
162            list.setVisible(false);
163        } else {
164            list.add(new AttributeModifier("rel", "tooltip"));
165            list.add(new AttributeModifier("title", filters.getTitle()));
166            list.setVisible(true);
167        }
168        setFiltersStatus(null);
169
170        // Choose type
171        RadioChoice<SearchType> typeSelect = new RadioChoice<SearchType>("type", new PropertyModel<SearchType>(pageStatusModel, "type"), types, new SearchTypeRenderer());
172        typeSelect.add(new AjaxFormChoiceComponentUpdatingBehavior() {
173            @Override
174            protected void onUpdate(AjaxRequestTarget target) {
175                SearchPage.this.addOrReplace(new EmptyPanel("boxes"));
176                getStatus().setSearch("");
177                target.add(search);
178                target.add(SearchPage.this.get("form").get("examplesContainer"));
179                target.add(SearchPage.this.get("boxes"));
180
181                if (getStatus().getType().getFilters() == null) {
182                    list.setVisible(false);
183                } else {
184                    list.setVisible(true);
185                }
186                setFiltersStatus(null);
187                target.add(list);
188            }
189        });
190        form.add(typeSelect);
191
192        add(form);
193
194        // Examples
195        WebMarkupContainer examples = new WebMarkupContainer("examplesContainer");
196        examples.setOutputMarkupId(true);
197        examples.add(new ListView<String>("examples", new ExamplesModel(new PropertyModel<SearchType>(pageStatusModel, "type"))) {
198
199            @Override
200            protected void populateItem(ListItem<String> item) {
201
202                AjaxLink<String> link = new AjaxLink<String>("link", item.getModel()) {
203                    @Override
204                    public void onClick(AjaxRequestTarget target) {
205                        getStatus().setSearch(getModelObject());
206                        ORI baseUri = SearchPage.this.getConfig().getWebsiteConfig().getORI().getParent();
207                        SearchPage.this.addOrReplace(new BoxesPanel("boxes", SearchPage.this.getStatus(), baseUri, null));
208                        target.add(search);
209                        target.add(SearchPage.this.get("boxes"));
210                    }
211                };
212
213                link.add(new Label("label", item.getModel()));
214                item.add(link);
215
216                WebMarkupContainer sep = new WebMarkupContainer("sep");
217                sep.setVisible(item.getIndex() + 1 != getModelObject().size());
218                item.add(sep);
219
220            }
221        });
222
223        form.add(examples);
224
225        ORI baseUri = SearchPage.this.getConfig().getWebsiteConfig().getORI().getParent();
226        add(new BoxesPanel("boxes", SearchPage.this.getStatus(), baseUri, null));
227
228    }

238    private Iterator<IEntity> getAutocompleteChoices(String input) {
239
240        Query query = new Query();
241        SearchType type = getStatus().getType();
242
243        ORI collectionUri = getAbsoluteUri(type.getCollection());
244        String collectionAlias = QueryUtils.newCollectionAlias(query, collectionUri);
245        query.setFrom(collectionAlias);
246
247        List<String> fieldList = type.getFieldsList();
248        query.addSelect(collectionAlias, fieldList);
249
250        for (String field : fieldList) {
251            QueryUtils.or(query, new Contains(collectionAlias, field, input));
252        }
253        query.addOrderBy(new OrderBy(collectionAlias, fieldList.get(0)));
254
255        query.setCount(10);
256
257        return new EntityIterator(collectionManager.load(query), collectionUri);
258    }

274    private class SearchTypeRenderer implements IChoiceRenderer<SearchType> {
275
276        @Override
277        public Object getDisplayValue(SearchType type) {
278
279            ORI collectionUri = type.getCollection();
280
281            Collection collection = getResourceManager().load(Collection.class, getAbsoluteUri(collectionUri));
282
283            if (collection == null) {
284                return collectionUri;
285            }
286
287            String title = collection.getTitle();
288            return (title == null ? collection.getName() : title);
289        }
290
291        @Override
292        public String getIdValue(SearchType object, int index) {
293            return Integer.toString(index);
294        }
295    }

297    /**
298     * Generic IEntity renderer that show all the fields.
299     */
300    private class EntityRenderer implements IAutoCompleteRenderer<IEntity> {
301
302        public final void render(final IEntity object, final Response response, final String criteria) {
303            String textValue = getTextValue(object, criteria);
304            if (textValue == null) {
305                throw new IllegalStateException(
306                        "A call to textValue(Object) returned an illegal value: null for object: " +
307                                object.toString());
308            }
309            textValue = textValue.replaceAll("\\\"", "&quot;");
310
311            response.write("<li textvalue=\"" + textValue + "\"");
312            response.write(">");
313            renderChoice(object, response, criteria);
314            response.write("</li>");
315        }
316
317        private String getTextValue(IEntity object, String criteria) {
318
319
320            SearchType type = getStatus().getType();
321            List<String> fields = type.getKeysList();
322
323            for (String field : fields) {
324                String value = String.valueOf(object.get(field));
325
326                if (StringUtils.containsIgnoreCase(value, criteria)) {
327                    return value;
328                }
329            }
330
331            return String.valueOf(object.get(fields.get(0)));
332        }
333
334        public final void renderHeader(final Response response) {
335            response.write("<ul>");
336        }
337
338        public final void renderFooter(final Response response, int count) {
339            response.write("</ul>");
340        }
341
342        protected void renderChoice(IEntity object, Response response, String criteria) {
343
344            SearchType type = getStatus().getType();
345            List<String> keys = type.getKeysList();
346            List<String> fields = type.getFieldsList();
347
348
349            boolean keyFieldAdded = false;
350            for (String field : fields) {
351                String value = String.valueOf(object.get(field));
352
353                if (StringUtils.containsIgnoreCase(value, criteria)) {
354                    if (keys.contains(field)) {
355                        keyFieldAdded = true;
356                    }
357
358                    renderValue(response, value, criteria, field);
359
360                }
361            }
362
363            if (!keyFieldAdded) {
364                String field = keys.get(0);
365                String value = String.valueOf(object.get(field));
366                renderValue(response, value, criteria, field);
367            }
368        }
369
370        private void renderValue(Response response, String value, String criteria, String field) {
371
372            String hlValue = value.replaceAll("(?i)(" + criteria + ")", "<strong>$1</strong>");
373            response.write("<span class='f'>" + field.toLowerCase() + ":</span>" + hlValue);
374            response.write("<br />");
375        }

381    private class ExamplesModel extends AbstractReadOnlyModel<List<String>> {
382
383        private IModel<SearchType> model;
384
385        private ExamplesModel(IModel<SearchType> model) {
386            this.model = model;
387        }
388
389        @Override
390        public List<String> getObject() {
391
392            SearchType searchType = model.getObject();
393
394            if (searchType == null || searchType.getExamples() == null) {
395                return Collections.EMPTY_LIST;
396            }
397
398            List<String> values = new ArrayList<String>();
399            for (String value : searchType.getExamples().split(",")) {
400                values.add(value.trim());
401            }
402            return values;
403        }
404    }
