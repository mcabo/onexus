018package org.onexus.website.api.widgets.filters;
019
020import org.apache.wicket.Component;
021import org.apache.wicket.ajax.AjaxRequestTarget;
022import org.apache.wicket.ajax.form.AjaxFormComponentUpdatingBehavior;
023import org.apache.wicket.ajax.markup.html.AjaxLink;
024import org.apache.wicket.event.Broadcast;
025import org.apache.wicket.markup.html.basic.Label;
026import org.apache.wicket.markup.html.form.ChoiceRenderer;
027import org.apache.wicket.markup.html.form.DropDownChoice;
028import org.apache.wicket.markup.html.form.Form;
029import org.apache.wicket.markup.html.list.ListItem;
030import org.apache.wicket.markup.html.list.ListView;
031import org.apache.wicket.markup.html.panel.EmptyPanel;
032import org.apache.wicket.model.IModel;
033import org.apache.wicket.model.PropertyModel;
034import org.onexus.website.api.events.EventAddFilter;
035import org.onexus.website.api.events.EventRemoveFilter;
036import org.onexus.website.api.pages.browser.BrowserPage;
037import org.onexus.website.api.pages.browser.IFilter;
038import org.onexus.website.api.widgets.Widget;
039import org.onexus.website.api.widgets.filters.custom.CustomFilter;
040import org.onexus.website.api.widgets.filters.custom.ListCustomFilterPanel;
041import org.onexus.website.api.widgets.filters.custom.NumericCustomFilterPanel;
042
043import java.util.List;
044
045public class FiltersWidget extends Widget<FiltersWidgetConfig, FiltersWidgetStatus> {
046
047    private CustomFilter currentFilter;
048
049    private final static Component EMPTY_CUSTOM_PANEL = new EmptyPanel("customPanel" ).setOutputMarkupId(true);
050
051    public FiltersWidget(String componentId, IModel<FiltersWidgetStatus> statusModel) {
052        super(componentId, statusModel);
053
054        // Filters list
055        final Form<String> filtersForm = new Form<String>("filtersForm" );
056        filtersForm.setOutputMarkupId(true);
057        filtersForm.add(new ListView<FilterConfig>("filters", new PropertyModel<List<? extends FilterConfig>>(statusModel, "filters" )) {
058
059            @Override
060            protected void populateItem(final ListItem<FilterConfig> item) {
061
062                final FilterConfig filterConfig = item.getModelObject();
063
064                item.setOutputMarkupId(true);
065
066                item.add(new AjaxLink<String>("remove" ) {
067
068                    @Override
069                    public void onClick(AjaxRequestTarget target) {
070                        FiltersWidget.this.getStatus().getFilters().remove(filterConfig);
071                        unapplyFilter(filterConfig, target);
072                        target.add(filtersForm);
073                    }
074
075                    @Override
076                    public boolean isVisible() {
077                        return filterConfig.isDeletable();
078                    }
079                });
080
081                // Add button
082                item.add(new AjaxLink<String>("apply" ) {
083
084                    @Override
085                    public void onClick(AjaxRequestTarget target) {
086                        applyFilter(filterConfig, target);
087                    }
088
089                    @Override
090                    public boolean isVisible() {
091                        return !isFilterApplyed(filterConfig);
092                    }
093
094                });
095
096                // Remove button
097                item.add(new AjaxLink<String>("unapply" ) {
098
099                    @Override
100                    public void onClick(AjaxRequestTarget target) {
101                        unapplyFilter(filterConfig, target);
102                    }
103
104                    @Override
105                    public boolean isVisible() {
106                        return isFilterApplyed(filterConfig);
107                    }
108                });
109
110                // Title
111                item.add(new Label("name", new TextFormaterPropertyModel(item.getModel(), "name", 65, true)));
112
113
114            }
115        });
116        add(filtersForm);
117
118        // Add custom filter select
119        Form<String> selectForm = new Form<String>("selectForm" );
120        DropDownChoice<CustomFilter> selector = new DropDownChoice<CustomFilter>(
121                "select",
122                new PropertyModel<CustomFilter>(this, "currentFilter" ),
123                getConfig().getCustomFilters(),
124                new ChoiceRenderer<CustomFilter>("title" ));
125
126        selector.add(new AjaxFormComponentUpdatingBehavior("onchange" ) {
127            @Override
128            protected void onUpdate(AjaxRequestTarget target) {
129
130                CustomFilter customFilter = getCurrentFilter();
131
132                if (customFilter == null || customFilter.getType() == null) {
133                    FiltersWidget.this.addOrReplace(EMPTY_CUSTOM_PANEL);
134                } else if (customFilter.getType().equalsIgnoreCase("list" )) {
135                    FiltersWidget.this.addOrReplace(new ListCustomFilterPanel("customPanel", customFilter) {
136                        @Override
137                        protected void addFilter(AjaxRequestTarget target, FilterConfig filterConfig) {
138                            addCustomFilter(target, filterConfig);
139                        }
140                    });
141                } else if (customFilter.getType().equalsIgnoreCase("numeric" )) {
142                    FiltersWidget.this.addOrReplace(new NumericCustomFilterPanel("customPanel", customFilter, FiltersWidget.this.getPageBaseOri()) {
143                        @Override
144                        protected void addFilter(AjaxRequestTarget target, FilterConfig filterConfig) {
145                            addCustomFilter(target, filterConfig);
146                        }
147                    });
148                }
149
150                target.add(FiltersWidget.this.get("customPanel" ));
151
152            }
153
154            private void addCustomFilter(AjaxRequestTarget target, FilterConfig filter) {
155                List<FilterConfig> filters = getStatus().getFilters();
156                filters.add(filter);
157                target.add(filtersForm);
158                applyFilter(filter, target);
159            }
160
161        });
162        selectForm.add(selector);
163        add(selectForm);
164
165        // Add custom filter panel
166        add(EMPTY_CUSTOM_PANEL);
167
168
169    }
170
171
172    protected boolean isFilterApplyed(FilterConfig filterConfig) {
173        List<IFilter> filters = getFilters();
174
175        for (IFilter filter : filters) {
176            if (filterConfig.equals(filter.getFilterConfig())) {
177                return true;
178            }
179        }
180        return false;
181    }
182
183    protected void unapplyFilter(FilterConfig filterConfig, AjaxRequestTarget target) {
184
185        List<IFilter> filters = getFilters();
186
187        IFilter removeMe = null;
188        for (IFilter filter : filters) {
189            if (filterConfig.equals(filter.getFilterConfig())) {
190                removeMe = filter;
191                break;
192            }
193        }
194
195        if (removeMe != null) {
196            filters.remove(removeMe);
197            send(getPage(), Broadcast.BREADTH, EventRemoveFilter.EVENT);
198        }
199    }
200
201    protected void applyFilter(FilterConfig filterConfig, AjaxRequestTarget target) {
202        List<IFilter> filters = getFilters();
203        filters.add(new BrowserFilter(filterConfig));
204        send(getPage(), Broadcast.BREADTH, EventAddFilter.EVENT);
205    }
206
207    public CustomFilter getCurrentFilter() {
208        return currentFilter;
209    }
210
211    public void setCurrentFilter(CustomFilter currentFilter) {
212        this.currentFilter = currentFilter;
213    }
214
215    protected List<IFilter> getFilters() {
216        return findParent(BrowserPage.class).getStatus().getFilters();
217    }
218}
