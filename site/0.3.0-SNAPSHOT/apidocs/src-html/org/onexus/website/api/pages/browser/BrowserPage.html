053public class BrowserPage extends Page<BrowserPageConfig, BrowserPageStatus> {
054
055    public BrowserPage(String componentId, IModel<BrowserPageStatus> statusModel) {
056        super(componentId, statusModel);
057        onEventFireUpdate(EventTabSelected.class, EventAddFilter.class, EventRemoveFilter.class);
058
059        add(new FiltersPanel("filters", statusModel));
060
061        add(new EmptyPanel("maps").setVisible(false));
062
063        onEventFireUpdate(EventViewChange.class);
064    }
065
066    protected boolean isCurrentTab(String tabId) {
067        return tabId.equals(getStatus().getCurrentTabId());
068    }
069
070    protected TabConfig getCurrentTab() {
071        String currentTabId = getStatus().getCurrentTabId();
072        return getConfig().getTab(currentTabId);
073    }
074
075    @Override
076    protected void onBeforeRender() {
077
078        VisibleTabs visibleTabs = new VisibleTabs();
079
080        // Tabs can change when we fix/unfix entities
081        addOrReplace(new ListView<TabGroup>("tabs", visibleTabs) {
082
083            @Override
084            protected void populateItem(ListItem<TabGroup> item) {
085
086                TabGroup tabGroup = item.getModelObject();
087
088                if (tabGroup.hasSubMenu()) {
089
090                    WebMarkupContainer link = new WebMarkupContainer("link");
091                    link.add(new AttributeModifier("class", "dropdown-toggle"));
092                    link.add(new AttributeModifier("data-toggle", "dropdown"));
093                    link.add(new AttributeModifier("href", "#"));
094                    link.add(new Label("label", tabGroup.getGroupLabel() + "<b class='caret'></b>").setEscapeModelStrings(false));
095
096                    item.add(link);
097
098                    if (tabGroup.containsTab(getStatus().getCurrentTabId())) {
099                        item.add(new AttributeModifier("class", "dropdown active"));
100                    } else {
101                        item.add(new AttributeModifier("class", "dropdown"));
102                    }
103
104                    WebMarkupContainer submenu = new WebMarkupContainer("submenu");
105                    submenu.add(new ListView<TabConfig>("item", tabGroup.getTabConfigs()) {
106
107                        @Override
108                        protected void populateItem(ListItem<TabConfig> item) {
109
110                            TabConfig tab = item.getModelObject();
111
112
113                            BrowserPageLink<String> link = new BrowserPageLink<String>("link", Model.of(tab.getId())) {
114
115                                @Override
116                                public void onClick(AjaxRequestTarget target) {
117                                    getStatus().setCurrentTabId(getModelObject());
118                                    sendEvent(EventTabSelected.EVENT);
119                                }
120
121                            };
122
123                            link.add(new Label("label", tab.getTitle()));
124                            item.add(link);
125
126                            if (isCurrentTab(tab.getId())) {
127                                item.add(new AttributeModifier("class", new Model<String>("active")));
128                            }
129
130                        }
131                    });
132                    item.add(submenu);
133
134                } else {
135
136                    TabConfig tab = tabGroup.getTabConfigs().get(0);
137                    BrowserPageLink<String> link = new BrowserPageLink<String>("link", Model.of(tab.getId())) {
138
139                        @Override
140                        public void onClick(AjaxRequestTarget target) {
141                            getStatus().setCurrentTabId(getModelObject());
142                            sendEvent(EventTabSelected.EVENT);
143                        }
144
145                    };
146
147                    link.add(new Label("label", tab.getTitle()));
148                    item.add(link);
149
150                    if (isCurrentTab(tab.getId())) {
151                        item.add(new AttributeModifier("class", new Model<String>("active")));
152                    }
153
154                    item.add(new WebMarkupContainer("submenu").setVisible(false));
155
156                }
157
158            }
159
160        });
161
162        // Check if the current selected tab is visible.
163        String currentTabId = getStatus().getCurrentTabId();
164
165        // Set a current tab if there is no one.
166        if (currentTabId == null) {
167
168            List<TabConfig> tabs = getConfig().getTabs();
169            if (tabs != null && !tabs.isEmpty()) {
170                currentTabId = tabs.get(0).getId();
171                getStatus().setCurrentTabId(currentTabId);
172
173                // Init currentPage
174                //TODO This is not a good solution
175                PageParameters pageParameters = getPage().getPageParameters();
176                pageParameters.add("ptab", currentTabId);
177                setResponsePage(Website.class, pageParameters);
178            }
179
180        }
181
182        List<TabGroup> tabs = visibleTabs.getObject();
183        boolean hiddenTab = true;
184        for (TabGroup tab : tabs) {
185            if (tab.containsTab(currentTabId)) {
186                hiddenTab = false;
187            }
188        }
189
190        if (hiddenTab && !tabs.isEmpty()) {
191            TabConfig firstTab = tabs.get(0).getTabConfigs().get(0);
192            getStatus().setCurrentTabId(firstTab.getId());
193        }
194
195
196        List<String> views = new ArrayList<String>();
197        if (getCurrentTab().getViews() != null) {
198            for (ViewConfig view : getCurrentTab().getViews()) {
199                views.add(view.getTitle());
200            }
201        }
202
203
204        if (getStatus().getCurrentView() == null && views.size() > 0) {
205            getStatus().setCurrentView(views.get(0));
206        }
207
208        // Check that the current view is visible
209        List<ViewConfig> filteredViews = new ArrayList<ViewConfig>();
210        VisiblePredicate predicate = new VisiblePredicate(getStatus().getORI().getParent(), getStatus().getFilters());
211        CollectionUtils.select(getCurrentTab().getViews(), predicate, filteredViews);
212
213        boolean visibleView = false;
214        String currentView = getStatus().getCurrentView();
215        for (ViewConfig view : filteredViews) {
216            if (view.getTitle().equals(currentView)) {
217                visibleView = true;
218                break;
219            }
220        }
221
222        if (!visibleView) {
223            getStatus().setCurrentView(filteredViews.get(0).getTitle());
224        }
225
226        ViewConfig viewConfig = null;
227        if (getCurrentTab().getViews() != null) {
228            for (ViewConfig view : getCurrentTab().getViews()) {
229                if (view.getTitle().equals(getStatus().getCurrentView())) {
230                    viewConfig = view;
231                    break;
232                }
233            }
234        }
235
236        if (viewConfig == null) {
237            addOrReplace(new EmptyPanel("content"));
238        } else {
239            if (viewConfig.getLeft() != null && viewConfig.getTop() != null) {
240                addOrReplace(new TopleftLayout("content", viewConfig, getModel()));
241            } else if (viewConfig.getLeft() != null) {
242                addOrReplace(new LeftMainLayout("content", viewConfig, getModel()));
243            } else if (viewConfig.getTop() != null || viewConfig.getTopRight() != null) {
244                addOrReplace(new TopmainLayout("content", viewConfig, getModel()));
245            } else {
246                addOrReplace(new SingleLayout("content", viewConfig, getModel()));
247            }
248        }
249
250        boolean visible = !isEmbed();
251        get("filters").setVisible(visible);
252        get("tabs").setVisible(visible);
253        //TODO viewSelector.setVisible(visible && views.size() > 1);
254
255        super.onBeforeRender();
256    }
257
258    private class VisibleTabs extends AbstractReadOnlyModel<List<TabGroup>> {
259
260        @Override
261        public List<TabGroup> getObject() {
262
263            List<TabConfig> allTabs = getConfig().getTabs();
264
265            // A predicate that filters the visible views
266            Predicate filter = new VisiblePredicate(getStatus().getORI().getParent(), getStatus().getFilters());
267
268            // Return a new collection with only the visible tabs
269            List<TabConfig> tabs = new ArrayList<TabConfig>();
270            CollectionUtils.select(allTabs, filter, tabs);
271
272            // Tab groups
273            List<TabGroup> tabGroups = new ArrayList<TabGroup>();
274            Map<String, TabGroup> subMenus = new HashMap<String, TabGroup>();
275
276            for (TabConfig tab : tabs) {
277                if (tab.getGroup() == null) {
278                    tabGroups.add(new TabGroup(tab));
279                } else {
280                    TabGroup group;
281                    if (!subMenus.containsKey(tab.getGroup())) {
282                        group = new TabGroup(tab.getGroup());
283                        subMenus.put(tab.getGroup(), group);
284                        tabGroups.add(group);
285                    } else {
286                        group = subMenus.get(tab.getGroup());
287                    }
288                    group.add(tab);
289                }
290            }
291
292            return tabGroups;
293        }
294
295    }
296
297    private static class TabGroup implements Serializable {
298
299        private String groupLabel;
300        private List<TabConfig> tabConfigs = new ArrayList<TabConfig>();
301
302        public TabGroup(String groupLabel) {
303            super();
304            this.groupLabel = groupLabel;
305        }
306
307        public TabGroup(TabConfig tab) {
308            super();
309            this.groupLabel = null;
310            this.tabConfigs.add(tab);
311        }
312
313        public boolean hasSubMenu() {
314            return (groupLabel != null);
315        }
316
317        public void add(TabConfig tabConfig) {
318            tabConfigs.add(tabConfig);
319        }
320
321        public boolean containsTab(String tabId) {
322            for (TabConfig tab : tabConfigs) {
323                if (tab.getId().equals(tabId)) {
324                    return true;
325                }
326            }
327            return false;
328        }
329
330        public String getGroupLabel() {
331            return groupLabel;
332        }
333
334        public List<TabConfig> getTabConfigs() {
335            return tabConfigs;
336        }
337    }
338
339    @Override
340    public void renderHead(IHeaderResponse response) {
341        super.renderHead(response);
342
343        if (isEmbed()) {
344            response.render(CssHeaderItem.forCSS("div.tab-row { display: none; }", "embed-browser-page"));
345        }
346    }
