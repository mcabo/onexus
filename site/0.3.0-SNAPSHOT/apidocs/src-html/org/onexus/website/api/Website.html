019

025import org.apache.wicket.markup.html.link.BookmarkablePageLink;
026import org.apache.wicket.markup.html.link.Link;
027import org.apache.wicket.markup.html.list.ListItem;
028import org.apache.wicket.markup.html.list.ListView;
029import org.apache.wicket.markup.html.panel.EmptyPanel;
030import org.apache.wicket.markup.html.panel.Panel;
031import org.apache.wicket.model.IModel;
032import org.apache.wicket.model.PropertyModel;
033import org.apache.wicket.request.mapper.parameter.PageParameters;
034import org.apache.wicket.util.string.StringValue;
035import org.onexus.resource.api.ORI;
036import org.onexus.website.api.pages.IPageManager;
037import org.onexus.website.api.pages.PageConfig;
038import org.onexus.website.api.pages.PageModel;
039import org.onexus.website.api.theme.DefaultTheme;
040import org.onexus.website.api.utils.CustomCssBehavior;
041import org.onexus.website.api.utils.HtmlDataResourceModel;
042import org.onexus.website.api.utils.authorization.Authorization;
043import org.onexus.website.api.utils.panels.ConnectionsPanel;
044import org.onexus.website.api.utils.panels.LoginPanel;
045import org.onexus.website.api.utils.panels.NotAuthorizedPage;
046import org.ops4j.pax.wicket.api.PaxWicketBean;
047
048import java.util.ArrayList;
049import java.util.List;
050
051public class Website extends WebPage {
052
053    // Parameters
054    public final static String PARAMETER_WEBSITE = "ori";
055    public final static String PARAMETER_PAGE = "c";
056
057    @PaxWicketBean(name = "pageManager")
058    private IPageManager pageManager;
059
060    public Website(PageParameters pageParameters) {
061        super(new WebsiteModel(pageParameters));
062
063        if (!Authorization.authorize(getConfig())) {
064            if (AuthenticatedWebSession.get().isSignedIn()) {
065                setResponsePage(NotAuthorizedPage.class);
066            } else {
067                WebsiteApplication.get().restartResponseAtSignInPage();
068            }
069        }
070
071        add(new DefaultTheme());
072
073        final WebsiteStatus status = getStatus();
074        final WebsiteConfig config = getConfig();
075
076        ORI parentUri = config.getORI().getParent();
077        add(new CustomCssBehavior(parentUri, config.getCss()));
078
079        // Init currentPage
080        if (status.getCurrentPage() == null || config.getPage(status.getCurrentPage()) == null) {
081            status.setCurrentPage(config.getPages().get(0).getId());
082        }
083
084        add(new Label("windowTitle", config.getTitle()));
085
086        add(new EmptyPanel("progressbar"));
087        //TODO add(new ProgressBar("progressbar", false));
088
089        String header = config.getHeader();
090
091        Label headerLabel = new Label("header", new HtmlDataResourceModel(parentUri, header));
092        headerLabel.setVisible(header != null && !header.isEmpty());
093        headerLabel.setEscapeModelStrings(false);
094        add(headerLabel);
095
096        WebMarkupContainer menuSection = new WebMarkupContainer("menuSection");
097        menuSection.add(new ListView<PageConfig>("menu", new PropertyModel<List<PageConfig>>(this, "pageConfigList" )) {
098
099            @Override
100            protected void populateItem(ListItem<PageConfig> item) {
101
102                PageConfig pageConfig = item.getModelObject();
103
104                PageParameters parameters = new PageParameters();
105                parameters.add(PARAMETER_PAGE, pageConfig.getId());
106
107                Link<String> link = new BookmarkablePageLink<String>("link", Website.class, parameters);
108                link.add(new Label("name", pageConfig.getLabel()));
109
110                String currentPage = status.getCurrentPage();
111
112                item.add(link);
113
114                if (currentPage.equals(pageConfig.getId())) {
115                    link.getParent().add(new AttributeModifier("class", "active" ));
116                }
117
118            }
119        });
120
121        if (pageParameters.get("embed").toBoolean(false)) {
122            menuSection.setVisible(false);
123        } else {
124            menuSection.setVisible(true);
125        }
126
127        // Login section
128        Panel login = new LoginPanel("login");
129        menuSection.add(login);
130        if (config.getLogin() == null || !config.getLogin()) {
131            login.setVisible(false);
132        }
133
134        // Projects section
135        menuSection.add( new ConnectionsPanel("connections", config.getConnections()) );
136
137        add(menuSection);
138
139        String currentPage = status.getCurrentPage();
140
141        add(pageManager.create("page", new PageModel(currentPage, (IModel<WebsiteStatus>) getDefaultModel())));
142
143        String bottom = config.getBottom();
144        Label bottomLabel = new Label("bottom", new HtmlDataResourceModel(parentUri, bottom));
145        bottomLabel.setVisible(bottom != null && !bottom.isEmpty());
146        bottomLabel.setEscapeModelStrings(false);
147        add(bottomLabel);
148
149
150    }
151
152    @Override
153    protected void onBeforeRender() {
154        StringValue embed = getPage().getPageParameters().get("embed");
155
156        boolean visible = !embed.toBoolean(false);
157
158        get("header").setVisible(visible);
159        get("bottom").setVisible(visible);
160
161        if (!Authorization.authorize(getConfig()) || !Authorization.authorize(getConfig().getPage(getStatus().getCurrentPage()))) {
162            if (AuthenticatedWebSession.get().isSignedIn()) {
163                setResponsePage(NotAuthorizedPage.class);
164            } else {
165                WebsiteApplication.get().restartResponseAtSignInPage();
166            }
167        }
168
169        super.onBeforeRender();
170    }
171
172    public WebsiteStatus getStatus() {
173        return (WebsiteStatus) getDefaultModelObject();
174    }
175
176    public WebsiteConfig getConfig() {
177        return getStatus().getConfig();
178    }
179
180    public List<PageConfig> getPageConfigList() {
181        List<PageConfig> pages = new ArrayList<PageConfig>();
182
183        for (PageConfig page : getConfig().getPages()) {
184
185            //TODO Manage optional show/hide non-authorized links
186            //if (Authorization.authorize(page)) {
187                pages.add(page);
188            //}
189        }
190
191        return pages;
192    }
