001package org.onexus.website.api.utils.panels;
002
003import org.apache.wicket.AttributeModifier;
004import org.apache.wicket.markup.html.WebMarkupContainer;
005import org.apache.wicket.markup.html.basic.Label;
006import org.apache.wicket.markup.html.link.ExternalLink;
007import org.apache.wicket.markup.repeater.RepeatingView;
008import org.apache.wicket.request.Url;
009import org.apache.wicket.request.mapper.parameter.PageParameters;
010import org.onexus.website.api.Connection;
011import org.onexus.website.api.Website;
012import org.onexus.website.api.WebsiteSession;
013import org.onexus.website.api.WebsiteStatus;
014import org.onexus.website.api.events.EventPanel;
015import org.onexus.website.api.events.EventQueryUpdate;
016
017import java.util.Collections;
018import java.util.List;
019
020public class ConnectionsPanel extends EventPanel {
021
022    private List<Connection> connections;
023
024    public ConnectionsPanel(String id, List<Connection> connections) {
025        super(id);
026
027        onEventFireUpdate(EventQueryUpdate.class);
028
029        if (connections == null) {
030            this.connections = Collections.emptyList();
031        } else {
032            this.connections = connections;
033        }
034
035        setVisible(!this.connections.isEmpty() && Boolean.parseBoolean(System.getProperty("org.onexus.website.connections", "false")));
036
037    }
038
039    @Override
040    protected void onBeforeRender() {
041        super.onBeforeRender();
042
043        String urlPath = "";
044        Website website = findParent(Website.class);
045        if (!connections.isEmpty()) {
046            PageParameters params = new PageParameters();
047
048            if (website != null) {
049                WebsiteStatus status = website.getStatus();
050                status.encodeParameters(params, true);
051            }
052
053            Url url = getRequestCycle().mapUrlFor(getPage().getClass(), params);
054            urlPath = url.getPath() + url.getQueryString();
055
056            int firstSlash = urlPath.indexOf('/');
057            urlPath = urlPath.substring(firstSlash);
058        }
059
060        RepeatingView connectionsView = new RepeatingView("projects");
061        for (Connection connection : connections) {
062            WebMarkupContainer connectionItem = new WebMarkupContainer(connectionsView.newChildId());
063            if (connection.getActive() != null && connection.getActive()) {
064                connectionItem.add(new AttributeModifier("class", "active"));
065            }
066            ExternalLink link = new ExternalLink("url", connection.getUrl() + urlPath);
067            link.add(new Label("title", connection.getTitle()));
068            connectionItem.add(link);
069            connectionsView.add(connectionItem);
070        }
071        addOrReplace(connectionsView);
072
073        WebMarkupContainer divider = new WebMarkupContainer("divider");
074        addOrReplace(divider);
075        RepeatingView userProjects = new RepeatingView("user");
076        addOrReplace(userProjects);
077
078        //TODO Use an user projects manager
079        if (WebsiteSession.get().getRoles().hasRole("tester")) {
080            WebMarkupContainer connectionItem = new WebMarkupContainer(userProjects.newChildId());
081            if (website.getConfig().getORI().getProjectUrl().endsWith("snp_hopkins_breast")) {
082                connectionItem.add(new AttributeModifier("class", "active"));
083            }
084            ExternalLink link = new ExternalLink("url", "/web/snp_hopkins_breast/website" + urlPath);
085            link.add(new Label("title", "Breast and colorectal cancer"));
086            connectionItem.add(link);
087            userProjects.add(connectionItem);
088        } else {
089            divider.setVisible(false);
090        }
091
092    }
093
094}
