057import java.io.IOException;

065public class DownloadWidget extends Widget<DownloadWidgetConfig, DownloadWidgetStatus> {
066
067    public final static ResourceReference CSS = new CssResourceReference(DownloadWidget.class, "prettify/prettify.css");
068    public final static ResourceReference JS = new JavaScriptResourceReference(DownloadWidget.class, "prettify/prettify.js");
069    private static Pattern commaPattern = Pattern.compile(",");
070
071    @PaxWicketBean(name = "collectionManager")
072    private ICollectionManager collectionManager;
073
074    public final static Map<String, IQueryScript> scriptsMap = new LinkedHashMap<String, IQueryScript>();
075    static {
076            addScript(new RScript());
077            addScript(new PythonScript());
078            addScript(new PerlScript());
079            addScript(new BashScript());
080    }
081    private static void addScript(IQueryScript queryScript) {
082        scriptsMap.put(queryScript.getLabel().toLowerCase(), queryScript);
083    }
084
085    private final static Map<String, IDownloadFormat> formatsMap = new LinkedHashMap<String, IDownloadFormat>();
086    static {
087            addFormat(new TsvFormat());
088            addFormat(new ExcelFormat());
089    }
090    private static void addFormat(IDownloadFormat format) {
091        formatsMap.put(format.getLabel().toLowerCase(), format);
092    }
093
094
095    private String webserviceUrl;
096    private IDownloadFormat format;
097
098    public DownloadWidget(String componentId, IModel<DownloadWidgetStatus> statusModel) {
099        super(componentId, statusModel);
100        onEventFireUpdate(EventQueryUpdate.class);
101
102        // Get query
103        final StringBuilder oql = new StringBuilder();
104        Query query = getQuery();
105        query.toString(oql, true);
106
107        // Webservice URL
108        String serviceMount = collectionManager.getMount();
109        webserviceUrl = WebsiteApplication.toAbsolutePath('/' + serviceMount);
110
111        // Download form
112        final Form<String> downloadForm = new Form<String>("form");
113        downloadForm.setOutputMarkupId(true);
114
115        List<IDownloadFormat> formats = getFormats();
116        if (formats.isEmpty()) {
117            throw new UnsupportedOperationException("Download widget: At least one format is needed.");
118        }
119        setFormat(formats.get(0));
120        downloadForm.add(new DropDownChoice<IDownloadFormat>("format", new PropertyModel<IDownloadFormat>(this, "format"), formats));
121        add(downloadForm);
122
123        final AjaxDownloadBehavior ajaxDownloadBehavior = new AjaxDownloadBehavior() {
124            @Override
125            protected String getFileName() {
126                return getFormat().getFileName();
127            }
128
129            @Override
130            protected IResource getResource() {
131                return new DownloadResource();
132            }
133        };
134        downloadForm.add(ajaxDownloadBehavior);
135
136        AjaxButton link = new AjaxButton("download") {
137
138            @Override
139            protected void onSubmit(AjaxRequestTarget target, Form<?> form) {
140                ajaxDownloadBehavior.initiate(target);
141                target.add(form);
142            }
143        };
144        downloadForm.add(link);
145
146        List<IQueryScript> scripts = getScripts();
147        // Add scripts
148        ListView scriptsView = new ListView<IQueryScript>("scripts", scripts) {
149            @Override
150            protected void populateItem(ListItem<IQueryScript> item) {
151
152                IQueryScript script = item.getModelObject();
153
154                // Code body
155                WebMarkupContainer body = new WebMarkupContainer("body");
156                body.setMarkupId(item.getMarkupId() + "-body");
157                item.add(body);
158                body.add(new Label("code", script.getContent(oql.toString(), webserviceUrl)).setEscapeModelStrings(false));
159
160                // Code toggle
161                Label toggle = new Label("toggle", "Use in " + script.getLabel() + " script");
162                toggle.add(new AttributeModifier("href", "#" + body.getMarkupId()));
163                item.add(toggle);
164
165            }
166        };
167
168        if (scripts.isEmpty()) {
169            scriptsView.setVisible(false);
170        }
171        add(scriptsView);
172
173    }
174
175    private List<IDownloadFormat> getFormats() {
176        List<IDownloadFormat> formats = new ArrayList<IDownloadFormat>();
177
178        String formatsStr = getConfig().getFormats();
179
180        if (formatsStr == null) {
181            formats.addAll(formatsMap.values());
182        } else {
183            String[] values = commaPattern.split(formatsStr);
184            for (String value : values) {
185                IDownloadFormat format = formatsMap.get(value.trim().toLowerCase());
186                if (format != null) {
187                    formats.add(format);
188                }
189            }
190        }
191
192        return formats;
193    }
194
195
196    private List<IQueryScript> getScripts() {
197        List<IQueryScript> scripts = new ArrayList<IQueryScript>();
198
199        String scriptsStr = getConfig().getScripts();
200
201        if (scriptsStr == null) {
202            scripts.addAll(scriptsMap.values());
203        } else {
204            String[] values = commaPattern.split(scriptsStr);
205            for (String value : values) {
206                IQueryScript script = scriptsMap.get(value.trim().toLowerCase());
207                if (script != null) {
208                    scripts.add(script);
209                }
210            }
211        }
212
213        return scripts;
214    }
215

218
219        if (event.getPayload() instanceof AjaxRequestTarget) {
220            AjaxRequestTarget target = (AjaxRequestTarget) event.getPayload();
221            target.appendJavaScript("prettyPrint();");
222        }
223
224    }

229        response.render(JavaScriptHeaderItem.forReference(JS));
230    }
231
232    public IDownloadFormat getFormat() {
233        return format;
234    }
235
236    public void setFormat(IDownloadFormat format) {
237        this.format = format;
238    }
239
240    private ICollectionManager getCollectionManager() {
241
242        if (collectionManager == null) {
243            WebsiteApplication.inject(this);
244        }
245
246        return collectionManager;
247    }
248
249    private class DownloadResource extends AbstractResource {
250
251        @Override
252        protected ResourceResponse newResourceResponse(Attributes attributes) {
253
254            ResourceResponse resourceResponse = new ResourceResponse();
255
256            resourceResponse.setContentDisposition(ContentDisposition.ATTACHMENT);
257            resourceResponse.setContentType(getFormat().getContentType());
258            resourceResponse.setFileName(getFormat().getFileName());
259            resourceResponse.setWriteCallback(new WriteCallback() {
260                @Override
261                public void writeData(Attributes attributes) throws IOException {
262
263                    Query query = getQuery();
264                    IDownloadFormat format = getFormat();
265
266                    if (format.getMaxRowsLimit() != null && (query.getCount() == null || format.getMaxRowsLimit() < query.getCount())) {
267                        query.setCount(format.getMaxRowsLimit());
268                    }
269
270                    IEntityTable table = getCollectionManager().load(query);
271                    OutputStream out = attributes.getResponse().getOutputStream();
272                    getFormat().write(table, out);
273                }
274            });
275
276            return resourceResponse;
277        }
278    }
