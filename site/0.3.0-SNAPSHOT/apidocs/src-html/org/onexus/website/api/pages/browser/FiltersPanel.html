018package org.onexus.website.api.pages.browser;
019
020import org.apache.wicket.AttributeModifier;
021import org.apache.wicket.ajax.AjaxRequestTarget;
022import org.apache.wicket.markup.html.WebMarkupContainer;
023import org.apache.wicket.markup.html.basic.Label;
024import org.apache.wicket.markup.repeater.RepeatingView;
025import org.apache.wicket.model.IModel;
026import org.apache.wicket.model.IWrapModel;
027import org.apache.wicket.model.Model;
028import org.onexus.collection.api.ICollectionManager;
029import org.onexus.collection.api.query.Query;
030import org.onexus.resource.api.IResourceManager;
031import org.onexus.resource.api.ORI;
032import org.onexus.website.api.WebsiteStatus;
033import org.onexus.website.api.events.EventAddFilter;
034import org.onexus.website.api.events.EventFiltersUpdate;
035import org.onexus.website.api.events.EventPanel;
036import org.onexus.website.api.events.EventRemoveFilter;
037import org.onexus.website.api.pages.PageStatus;
038import org.ops4j.pax.wicket.api.PaxWicketBean;
039
040import java.util.List;
041
042public class FiltersPanel extends EventPanel {
043
044    @PaxWicketBean(name = "resourceManager")
045    public transient IResourceManager resourceManager;
046
047    @PaxWicketBean(name = "collectionManager")
048    public transient ICollectionManager collectionManager;
049
050
051    public FiltersPanel(String id, IModel<BrowserPageStatus> pageModel) {
052        super(id, pageModel);
053
054        // Update this component if this events are fired.
055        onEventFireUpdate(EventAddFilter.class, EventRemoveFilter.class, EventFiltersUpdate.class);
056    }
057
058    @Override
059    protected void onBeforeRender() {
060        super.onBeforeRender();
061
062        RepeatingView filterRules = new RepeatingView("filter");
063
064        List<IFilter> filters = getBrowserPage().getFilters();
065
066        if (filters != null && !filters.isEmpty()) {
067
068            Query query = getQuery();
069
070            for (int i = 0; i < filters.size(); i++) {
071
072                IFilter filter = filters.get(i);
073
074                WebMarkupContainer container = new WebMarkupContainer(filterRules.newChildId());
075
076                // Add new fixed entity
077                //TODO container.add(new Label("collectionLabel", filter.getLabel(query)));
078                Label labelComponent = new Label("title", filter.getTitle(query));
079                labelComponent.setEscapeModelStrings(false);
080                container.add(labelComponent);
081
082                //TODO container.add(filter.getTooltip("box", query));
083
084                BrowserPageLink<Integer> removeLink = new BrowserPageLink<Integer>("remove", Model.of(i)) {
085
086                    @Override
087                    public void onClick(AjaxRequestTarget target) {
088                        Integer pos = getModelObject();
089                        getBrowserPageStatus().getFilters().remove(pos.intValue());
090                        sendEvent(EventRemoveFilter.EVENT);
091                    }
092
093                };
094
095                if (filter.isEnable()) {
096                    container.add(new AttributeModifier("class", "btn btn-large"));
097                } else {
098                    container.add(new AttributeModifier("class", "btn btn-large disabled"));
099                }
100
101                container.add(removeLink);
102                filterRules.add(container);
103            }
104        }
105
106        addOrReplace(filterRules);
107
108    }
109
110    private BrowserPageStatus getBrowserPage() {
111        return (BrowserPageStatus) getDefaultModelObject();
112    }
113
114    protected Query getQuery() {
115        PageStatus pageStatus = findParentStatus(getDefaultModel(), PageStatus.class);
116        return (pageStatus == null ? null : pageStatus.buildQuery(getBaseUri()));
117    }
118
119    protected ORI getBaseUri() {
120        WebsiteStatus websiteStatus = findParentStatus(getDefaultModel(), WebsiteStatus.class);
121        return (websiteStatus == null ? null : websiteStatus.getConfig().getORI().getParent());
122    }
123
124    private static <T> T findParentStatus(IModel<?> model, Class<T> statusClass) {
125
126        Object obj = model.getObject();
127
128        if (obj != null && statusClass.isAssignableFrom(obj.getClass())) {
129            return (T) obj;
130        }
131
132        if (model instanceof IWrapModel) {
133            IModel<?> parentModel = ((IWrapModel) model).getWrappedModel();
134            return findParentStatus(parentModel, statusClass);
135        }
136
137        return null;
138    }
139
140}
