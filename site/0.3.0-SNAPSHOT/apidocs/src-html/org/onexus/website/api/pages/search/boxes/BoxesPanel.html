018package org.onexus.website.api.pages.search.boxes;
019
020import org.apache.wicket.AttributeModifier;
021import org.apache.wicket.markup.html.basic.Label;
022import org.apache.wicket.markup.html.panel.Panel;
023import org.apache.wicket.markup.repeater.RepeatingView;
024import org.onexus.collection.api.ICollectionManager;
025import org.onexus.collection.api.IEntityTable;
026import org.onexus.collection.api.query.Contains;
027import org.onexus.collection.api.query.OrderBy;
028import org.onexus.collection.api.query.Query;
029import org.onexus.collection.api.utils.QueryUtils;
030import org.onexus.resource.api.ORI;
031import org.onexus.website.api.pages.search.SearchLink;
032import org.onexus.website.api.pages.search.SearchPageStatus;
033import org.onexus.website.api.pages.search.SearchType;
034import org.onexus.website.api.widgets.filters.FilterConfig;
035import org.ops4j.pax.wicket.api.PaxWicketBean;
036
037import java.util.List;
038
039public class BoxesPanel extends Panel {
040
041    @PaxWicketBean(name = "collectionManager")
042    private ICollectionManager collectionManager;
043
044    public BoxesPanel(String id, SearchPageStatus status, ORI baseUri, FilterConfig filterConfig) {
045        super(id);
046        setMarkupId("boxes");
047        add(new AttributeModifier("class", "accordion"));
048
049        RepeatingView boxes = new RepeatingView("boxes");
050
051        SearchType type = status.getType();
052
053        if (status.getSearch() == null) {
054            List<SearchLink> links = type.getFixLinks();
055            if (links != null && !links.isEmpty()) {
056                boxes.add(new MainLinksBox(boxes.newChildId(), links));
057            }
058        } else {
059            ORI collectionUri = type.getCollection().toAbsolute(baseUri);
060            if (filterConfig == null) {
061
062                IEntityTable table = getEntityTable(type, collectionUri, status.getSearch());
063
064
065                int count = 0;
066                while (table.next()) {
067                    boxes.add(new EntitySelectBox(boxes.newChildId(), count, status, table.getEntity(collectionUri)));
068                    count++;
069                }
070
071                if (count == 0) {
072                    boxes.add(new Label(boxes.newChildId(), "No results found").add(new AttributeModifier("class", "alert")));
073                }
074
075            } else {
076                boxes.add(new EntitySelectBox(boxes.newChildId(), 0, status, collectionUri, filterConfig));
077            }
078        }
079
080
081        add(boxes);
082
083    }
084
085    private IEntityTable getEntityTable(SearchType type, ORI collectionUri, String search) {
086
087        Query query = new Query();
088
089        String collectionAlias = QueryUtils.newCollectionAlias(query, collectionUri);
090        query.setFrom(collectionAlias);
091
092        List<String> fieldList = type.getFieldsList();
093        query.addSelect(collectionAlias, null);
094
095        for (String field : fieldList) {
096            QueryUtils.or(query, new Contains(collectionAlias, field, search));
097        }
098
099        query.addOrderBy(new OrderBy(collectionAlias, fieldList.get(0)));
100        query.setCount(20);
101
102        return collectionManager.load(query);
103    }
104

106}
