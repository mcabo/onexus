001package org.onexus.website.api;
002
003import org.apache.wicket.request.mapper.parameter.PageParameters;
004import org.apache.wicket.util.tester.WicketTester;
005import org.junit.Before;
006import org.junit.Test;
007import org.onexus.resource.api.*;
008import org.onexus.resource.serializer.xstream.internal.ResourceSerializer;
009import org.onexus.ui.authentication.jaas.JaasSignInPage;
010import org.onexus.website.api.mocks.MockPaxWicketInjector;
011import org.onexus.website.api.mocks.MockResourceManager;
012import org.onexus.website.api.pages.DefaultPageManager;
013import org.onexus.website.api.pages.IPageManager;
014import org.onexus.website.api.pages.browser.BrowserPageCreator;
015import org.onexus.website.api.widgets.DefaultWidgetManager;
016import org.onexus.website.api.widgets.IWidgetManager;
017import org.onexus.website.api.widgets.text.TextWidgetCreator;
018import org.ops4j.pax.wicket.api.InjectorHolder;
019
020import java.util.Arrays;
021
022public class WebsiteTest {
023
024    private WicketTester tester;
025
026    @Before
027    public void setUp() {
028
029        // Create beans
030        IResourceSerializer resourceSerializer = new ResourceSerializer();
031        resourceSerializer.register(WebsiteConfig.class);
032
033        IResourceRegister resourceRegister = new DefaultResourceRegister(resourceSerializer);
034        IResourceManager resourceManager = new MockResourceManager();
035
036        IPageManager pageManager = new DefaultPageManager(
037                Arrays.asList(
038                    new BrowserPageCreator()
039                ),
040                resourceRegister
041        );
042
043        IWidgetManager widgetManager = new DefaultWidgetManager(
044                Arrays.asList(
045                        new TextWidgetCreator()
046                ),
047                resourceRegister
048        );
049
050        // Create a Website resource
051        WebsiteConfig website = resourceSerializer.unserialize(WebsiteConfig.class, getClass().getResourceAsStream("test1.onx"));
052        website.setORI(new ORI("http://localhost/test?website"));
053        resourceManager.save(website);
054
055        // Create and initialize a Wicket application
056        WebsiteApplication application = new WebsiteApplication();
057        application.setWebsiteOri(website.getORI());
058        application.setWebPath(website.getName());
059        application.setSignInPageClass(JaasSignInPage.class);
060
061        tester = new WicketTester(application);
062
063        // Create an injector
064        MockPaxWicketInjector injector = new MockPaxWicketInjector();
065        injector.addBean(IResourceManager.class, resourceManager);
066        injector.addBean(IPageManager.class, pageManager);
067        injector.addBean(IWidgetManager.class, widgetManager);
068
069        InjectorHolder.setInjector(application.getApplicationKey(), injector);
070        tester.getApplication().getComponentInstantiationListeners().add(injector);
071
072    }
073
074    @Test
075    public void renderPage(){
076        tester.startPage(new Website(new PageParameters()));
077        tester.assertRenderedPage(Website.class);
078        //tester.dumpPage();
079    }
080
081}
